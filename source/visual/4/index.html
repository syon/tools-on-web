All Colors of 16x16x16

d3.select("body").selectAll("span")
    .data( make_data() )
  .enter().append("span")
    .classed({'box': true})
    .text(function(d) { return "#" + d; })
    .style({"background": function(d) { return "#" + d; }})
  ;

function make_data() {
    var data = new Array();
    for (var i=0; i<16*16*16; i++) {
        data.push( ("000"+(i).toString(16)).slice(-3) );
    }
    return data;
}
